Fitting a Servo's Model Parameters to Experimental Data

Import required packages and modules

import matplotlib.pyplot as plt
import numpy
import math
import yaml
import mujoco
import numpy
import mediapy as media

Create a square-wave function for emulating the control signal sent to the pwm model

def square(t,A,f,w,b,t_0):
    y = (t-t_0)*f 
    y = y%1
    y = (y<w)*1
    y = A*y +b
    return y

Motor Parameters were taken from the web, the rest caculated using steady-state motor equations

Vnom = 6
R = Vnom/.600
G = 55.5
t_stall = 15/100/G
i_stall = .6
i_nl = .2
w_nl = .66*1000*2*math.pi/180*G

kt = t_stall/ i_stall
# kv= Vnom/w_nl
ke = kt

b = kt*i_nl/w_nl

ts = 1e-4

Pull the servo's "massaged" experimental data.

with open('servo_data_collection.yml') as f:
    servo_data = yaml.load(f,Loader=yaml.Loader)

Convert to numpy, check the shape

t_data = numpy.array(servo_data['t'])

determine the timestep of the data

dt_data = (t_data[-1]-t_data[0])/len(t_data)

convert the theta data to a compatible numpy array and check the shape. needs to match mujoco's simulation data

q_data = numpy.array([servo_data['theta_u']]).T

create the desired control signal (just for plotting)

desired = square(t_data,A=servo_data['A'],f = servo_data['f'],w=servo_data['w'],b=servo_data['b'],t_0=servo_data['t_0'])

define my xml template and format, supplying the time constant from above variables

xml_template = """
<mujoco>
    <option timestep="{ts}"/>
    <!--
    <option integrator="Euler"/>
    <option tolerance="1e-8"/>
    <option solver="CG"/>
    -->
    <compiler angle="degree" />

    <worldbody>
        <light name="top" pos="0 0 10"/>
        <body name="body_1" pos="0 0 0" axisangle="1 0 0 0">
            <joint name="joint_1" type="hinge" axis="1 0 0" pos="0 0 0"/>
            <geom type="cylinder" size=".00725 .024" pos="0 0 0" rgba="0 1 1 1" mass=".016"/>
            <geom type="box" size=".005 .005 .005" pos="0 0 .024" rgba="0 1 1 1" mass="0"/>
        </body>
    </worldbody>
<actuator>
    <motor name="motor_1" joint="joint_1"/>
</actuator>
</mujoco>
"""

xml = xml_template.format(ts = ts)

create my model, data and renderer and a function that runs the model, changing the kp and b values as needed.

# xml = xml_template.format()

model = mujoco.MjModel.from_xml_string(xml)
data = mujoco.MjData(model)
renderer = mujoco.Renderer(model)

def run_sim(kp,b_act,render=False):

    # kp = 100
    V_control = 5
    # b_act = b*5

    def mycontroller1(model, data):
        w = data.qvel[0]
        actual = data.qpos[0]

        t = data.time
        desired = square(t,A=servo_data['A'],f = servo_data['f'],w=servo_data['w'],b=servo_data['b'],t_0=servo_data['t_0'])
        error = desired-actual
        V = kp*error
        if V>V_control: V=V_control
        if V<-V_control: V=-V_control
        torque = (kt*(V-(ke)*w*G)/R-b_act*w*G)*G

        data.ctrl = [torque]
        return

    try:
        mujoco.set_mjcb_control(mycontroller1)

        duration = t_data[-1]  # (seconds)
        framerate = 30  # (Hz)
        data_rate = 1/dt_data

        q = []
        w = []
        t = []
        frames = []

        mujoco.mj_resetData(model, data)

        while data.time < duration:
            # print(data.time)
            mujoco.mj_step(model, data)
            if len(frames) < data.time * framerate:
                renderer.update_scene(data)
                pixels = renderer.render()
                frames.append(pixels)
            if len(t) < data.time * data_rate:
                # print(data.time)
                q.append(data.qpos.copy())
                w.append(data.qvel.copy())
                t.append(data.time)
        if render:
            media.show_video(frames, fps=framerate)

        mujoco.set_mjcb_control(None)

        t = numpy.array(t)
        q = numpy.array(q)
        q = q[:len(q_data)]

    except:

        mujoco.set_mjcb_control(None)

    return t, q

Run the model and output the time / joint values

t,q = run_sim(kp = 11,b_act = b,render=True)

import scipy.optimize and a function that your minimize function will call. It will run the sim and compare the sim data from the experimental data, find the error, and return the sum of squared error over time

import scipy.optimize as so

def fun(vars):

    k,b = vars
    t,q = run_sim(k,b)


    error = q-q_data
    error = error**2
    error = error.sum()
    error = error**.5
    print(k,b,error)
    return error

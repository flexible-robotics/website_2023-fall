    import os
import mujoco
import numpy
import mediapy as media
import matplotlib.pyplot as plt
import idealab_tools.units as units

    xml_template = """
<mujoco>
    <option gravity="0 0 -{g:e}" />
    <worldbody>
        <light name="top" pos="0 0 {l:e}"/>
        <body name="A" pos="0 0 0">
            <joint name="j1" type="hinge" axis="0 1 0" pos="0 0 0" stiffness="{k:e}" damping="{b:e}"/>
            <geom type="box" size="{l2:e} {w2:e} {w2:e}" pos="{l2:e} 0 0" rgba="1 0 0 1"/>
            <inertial pos="{l2:e} 0 0" mass="{m:e}" diaginertia="{ixx:e} {iyy:e} {izz:e}" />
            <body name="B" pos="{l:e} 0 0">
                <joint name="j2" type="hinge" axis="0 1 0" pos="0 0 0" stiffness="{k}" damping="{b}"/>
                <geom type="box" size="{l2:e} {w2:e} {w2:e}" pos="{l2:e} 0 0" rgba="1 0 0 1"/>
                <inertial pos="{l2:e} 0 0" mass="{m:e}" diaginertia="{ixx:e} {iyy:e} {izz:e}"/>
            </body>
        </body>
    </worldbody>
</mujoco>
"""

    units.Unit.set_scaling(meter=1)

g = 9.81*units.acceleration
k = 100*units.rotational_stiffness
b = 1*(units.torque/units.rotational_speed)
m = 1*units.mass
l = 1*units.length
w = .1*units.length
ixx = 1*units.inertia
iyy = 1*units.inertia
izz = 1*units.inertia
s1 = 1*units.length

    xml = xml_template.format(k=k,b=b,m=m,l=l,l2=l/2,w2 = w/2,ixx = ixx,iyy=iyy,izz=izz,g=g)
print(xml)

    model = mujoco.MjModel.from_xml_string(xml)
data = mujoco.MjData(model)
renderer = mujoco.Renderer(model)

    duration = 10  # (seconds)
framerate = 30  # (Hz)

    xyz = []
t = []
frames = []
mujoco.mj_resetData(model, data)

while data.time < duration:
    mujoco.mj_step(model, data)
    xyz.append(data.xpos.copy())
    t.append(data.time)
    if len(frames) < data.time * framerate:
        renderer.update_scene(data)
        pixels = renderer.render()
        frames.append(pixels)

xyz = numpy.array(xyz)

    media.show_video(frames, fps=framerate)
